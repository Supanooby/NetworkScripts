Router1
---------------
en
conf t
hostname R1
no ip domain-lookup

enable secret class
line con 0
password cisco 
login
line vty 0 15 
password cisco
login
exit
service password-encryption

banner motd #This is R1. Unauthorized access is denied#
do wr



inte G0/0/1 
ip address 10.53.0.1 255.255.255.0
no shut

inte Loopback1
ip address 172.16.1.1 255.255.255.0
no shut

router ospf 56
router-id 1.1.1.1
network 10.53.0.0 0.0.0.255 area 0


!OSPF PRIORITY
inte g0/0/1	
ip ospf priority 50


!Hello timers
inte g0/0/1
ip ospf hello-interval 30
exit


!Static Route
ip route 0.0.0.0 0.0.0.0 Loopback1
router ospf 56 
default-information originate
exit

!Change Reference Bandwidth
router ospf 56
auto-cost reference-bandwidth 1000
exit


!Restart OSPF
clear ip ospf process 
yes


!Verify 
show ip ospf interface g0/0/1 
show ip route ospf 
show ip route ospf command. 















R2
----------------------------
en
conf t
hostname R2
no ip domain-lookup

enable secret class
line con 0
password cisco 
login
line vty 0 15 
password cisco
login
exit
service password-encryption

banner motd #This is R2. Unauthorized access is denied#
do wr



inte g0/0/1 
ip address 10.53.0.1 255.255.255.0
no shut

inte Loopback1 
ip address 192.168.1.1 255.255.255.0
no shut



router ospf 56
router-id 2.2.2.2
network 10.53.0.0 0.0.0.255 area 0
network 192.168.1.0 0.0.0.255 area 0


!Hello Timers
inte g0/0/1
ip ospf hello-interval 30

!Make interface Point-to-Point network
inte loopback1
ip ospf network point-to-point

!Passive Interface (Disable Router Advertisements being sent on Interface)
router ospf 56
passive-interface loopback1
do show ip protocools



!Change Reference Bandwidth
router ospf 56
auto-cost reference-bandwidth 1000
exit


!Restart OSPF
clear ip ospf process 
yes
